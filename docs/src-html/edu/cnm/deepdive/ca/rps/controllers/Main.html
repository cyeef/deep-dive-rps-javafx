001package edu.cnm.deepdive.ca.rps.controllers;
002
003import edu.cnm.deepdive.ca.rps.util.Constants;
004import java.util.ResourceBundle;
005import javafx.application.Application;
006import javafx.fxml.FXMLLoader;
007import javafx.scene.Parent;
008import javafx.scene.Scene;
009import javafx.stage.Stage;
010import javax.annotation.Resource;
011
012public class Main extends Application {
013
014
015  @Override
016  public void start(Stage primaryStage) throws Exception {
017
018    ResourceBundle bundle = ResourceBundle.getBundle(Constants.RESOURCES_UI_STRINGS);
019    FXMLLoader loader = new FXMLLoader(
020        getClass().getClassLoader().getResource(Constants.RESOURCES_MAIN_FXML), bundle);
021
022    Parent root = loader.load();
023    Controller controller = loader.getController();
024    controller.setBundle(bundle);
025
026    primaryStage.setTitle(bundle.getString(Constants.WINDOW_TITLE));
027    primaryStage.setResizable(false);
028    primaryStage.setScene(new Scene(root, Constants.WINDOW_WIDTH, Constants.NEW_WINDOW_HEIGHT));
029    primaryStage.show();
030  }
031
032
033  public static void main(String[] args) {
034    launch(args);
035  }
036}
